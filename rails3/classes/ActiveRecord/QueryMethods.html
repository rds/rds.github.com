186:       return if args.blank?
187: 
188:       builder = PredicateBuilder.new(table.engine)
189: 
190:       conditions = if [String, Array].include?(args.first.class)
191:         @klass.send(:sanitize_sql, args.size > 1 ? args : args.first)
192:       elsif args.first.is_a?(Hash)
193:         attributes = @klass.send(:expand_hash_conditions_for_aggregates, args.first)
194:         builder.build_from_hash(attributes, table)

197:       end
